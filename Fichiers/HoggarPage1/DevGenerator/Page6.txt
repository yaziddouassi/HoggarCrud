    <div class="w-full max-w-md bg-[#111] p-8 rounded-xl shadow">
      
      <!-- Titre Company -->
      <h1 class="text-xl font-semibold text-center text-blue-600 mb-2 uppercase tracking-wide">
        {{ props.company }}
      </h1>

      <!-- Formulaire avec autocomplete désactivé -->

            id="email"
            autocomplete="off"
            class="w-full h-[44px] px-4 py-2 border border-gray-600 rounded-[2px] bg-[#222] text-gray-200 focus:outline-none focus:ring focus:ring-white"
            required
            autofocus
          />
        </div>

        <div class="mb-4">
          <label class="block text-white mb-2" for="password">Password</label>
          <input
            v-model="form.password"
            type="password"
            id="password"
            autocomplete="new-password"
            class="w-full h-[44px] px-4 py-2 border border-gray-600 rounded-[2px] bg-[#222] text-gray-200 focus:outline-none focus:ring focus:ring-white"
            required
          />
        </div>

        <!-- Case à cocher Remember Me -->
        <div class="mb-6 flex items-center">
          <input
            v-model="form.remember"
            type="checkbox"
            id="remember"
            class="h-4 w-4 text-blue-600 bg-gray-700 border-gray-600"
